Install node.js:
sudo apt-get install curl software-properties-common
curl -sL https://deb.nodesource.com/setup_10.x | sudo bash -
sudo apt-get install nodejs
sudo npm install -g browserify

install packages:
npm install browserify ethereumjs-util buffer merkle-patricia-tree ethereumjs-wallet

create embeded browser file:
browserify main.js -o website_index/js/HeliosTxBundle.js
